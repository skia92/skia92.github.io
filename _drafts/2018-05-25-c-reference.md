# Data type
* u8 = uint8_t: unsigned char = 1 byte = 8 bits
* u16 = uint16_t: unsigned short
* u32 = uint32_t: unsigned int
* u64 = uint64_t: unsigned long long
* s8 = int8_t: signed char
* s16 = int16_t: short
* s32 = int32_t: int
* s64 = int64_t: long long
